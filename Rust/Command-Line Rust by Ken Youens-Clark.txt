Linux-Commands
==============

file hello   // show the kind of file

tree		 // show the directory structure

true         // set 0 as exit status
false        // set 1 as exit status

echo $?      // show exit status of the previous command

echo $PATH | tr : '\n'   // show paths in an formatted way
